Usage (Host Integration Pattern)

const el = document.querySelector('banking-search');
let controller;

el.filters = [
  { id: 'all', label: 'All' },
  { id: 'accounts',     label: 'Accounts' },
  { id: 'transactions', label: 'Transactions' },
  { id: 'customers',    label: 'Customers' },
];

el.addEventListener('bs:search', async ({ detail }) => {
  controller?.abort();
  controller = new AbortController();
  el.setAttribute('loading', '');
  el.removeAttribute('error');
  try {
    const data = await fetchResults(detail.term, detail.filter, controller.signal);
    el.results = data;
  } catch (err) {
    if (err.name === 'AbortError') return;
    el.setAttribute('error', classifyError(err)); // 'network' | 'timeout' | ...
  } finally {
    el.removeAttribute('loading');
  }
});

el.addEventListener('bs:select', ({ detail }) => router.navigate(detail.item.url));
el.addEventListener('bs:retry',  ({ detail }) => el.dispatchEvent(new CustomEvent('bs:search', { detail })));
